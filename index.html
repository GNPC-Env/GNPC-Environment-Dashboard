This page is for the platform administrator. It contains the Supabase database setup SQL and Row Level Security configuration needed to make data entry fully operational.
Step 1 — Create Supabase Tables (run in SQL Editor)
-- ══ PETROLEUM HOUSE ENERGY ══
CREATE TABLE petroleum_house_energy (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year            integer NOT NULL,
  month           text NOT NULL,
  consumption_kwh numeric,
  cost_ghc        numeric,
  tariff_ghc_kwh  numeric,
  co2e_tons       numeric,
  notes           text,
  entered_by      text,
  created_at      timestamptz DEFAULT now()
);

-- ══ PETROLEUM HOUSE WASTE ══
CREATE TABLE petroleum_house_waste (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year        integer NOT NULL,
  month       text NOT NULL,
  general_kg  numeric,
  plastic_kg  numeric,
  location    text,
  notes       text,
  entered_by  text,
  created_at  timestamptz DEFAULT now()
);

-- ══ MANAGED WASTE COLLECTIONS ══
CREATE TABLE managed_waste_collections (
  id             uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  week_ending    date NOT NULL,
  pickups        integer,
  food_kg        numeric,
  packaging_kg   numeric,
  residual_kg    numeric,
  total_kg       numeric,
  diversion_rate text,
  notes          text,
  entered_by     text,
  created_at     timestamptz DEFAULT now()
);

-- ══ PETROLEUM HOUSE WATER ══
CREATE TABLE petroleum_house_water (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year        integer NOT NULL,
  month       text NOT NULL,
  volume_m3   numeric,
  cost_ghc    numeric,
  source      text,
  notes       text,
  entered_by  text,
  created_at  timestamptz DEFAULT now()
);

-- ══ UPSTREAM FIELD DATA ══
CREATE TABLE upstream_field_data (
  id            uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  field         text NOT NULL,
  year          integer NOT NULL,
  month         text NOT NULL,
  ghg_tco2e     numeric,
  flare_total   numeric,
  flare_avg     numeric,
  pw_avg_mgl    numeric,
  pw_peak_mgl   numeric,
  oil_bblsd     numeric,
  gas_mmscfd    numeric,
  entered_by    text,
  created_at    timestamptz DEFAULT now()
);

-- ══ PETROLEUM HOUSE FLEET ══
CREATE TABLE petroleum_house_fleet (
  id            uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year          integer NOT NULL,
  month         text NOT NULL,
  vehicle       text,
  fuel_type     text,
  litres        numeric,
  distance_km   numeric,
  co2e_tco2e    numeric,
  notes         text,
  entered_by    text,
  created_at    timestamptz DEFAULT now()
);

-- ══ PETROLEUM HOUSE TRAVEL (AIR) ══
CREATE TABLE petroleum_house_travel (
  id            uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  year          integer NOT NULL,
  month         text NOT NULL,
  route         text,
  flight_class  text,
  passengers    integer,
  distance_km   numeric,
  haul_type     text,
  co2e_tco2e    numeric,
  notes         text,
  entered_by    text,
  created_at    timestamptz DEFAULT now()
);
Step 2 — Enable Row Level Security (RLS)
-- Enable RLS on all tables
ALTER TABLE petroleum_house_energy      ENABLE ROW LEVEL SECURITY;
ALTER TABLE petroleum_house_waste       ENABLE ROW LEVEL SECURITY;
ALTER TABLE managed_waste_collections   ENABLE ROW LEVEL SECURITY;
ALTER TABLE petroleum_house_water       ENABLE ROW LEVEL SECURITY;
ALTER TABLE upstream_field_data         ENABLE ROW LEVEL SECURITY;

-- Allow authenticated users to SELECT (read) all rows
CREATE POLICY "Auth users can read energy"    ON petroleum_house_energy      FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can read waste"     ON petroleum_house_waste       FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can read managed"   ON managed_waste_collections   FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can read water"     ON petroleum_house_water       FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can read upstream"  ON upstream_field_data         FOR SELECT TO authenticated USING (true);

-- Allow authenticated users to INSERT rows
CREATE POLICY "Auth users can insert energy"   ON petroleum_house_energy      FOR INSERT TO authenticated WITH CHECK (true);
CREATE POLICY "Auth users can insert waste"    ON petroleum_house_waste       FOR INSERT TO authenticated WITH CHECK (true);
CREATE POLICY "Auth users can insert managed"  ON managed_waste_collections   FOR INSERT TO authenticated WITH CHECK (true);
CREATE POLICY "Auth users can insert water"    ON petroleum_house_water       FOR INSERT TO authenticated WITH CHECK (true);
CREATE POLICY "Auth users can insert upstream" ON upstream_field_data         FOR INSERT TO authenticated WITH CHECK (true);

-- Fleet and Travel tables
ALTER TABLE petroleum_house_fleet   ENABLE ROW LEVEL SECURITY;
ALTER TABLE petroleum_house_travel  ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Auth users can read fleet"    ON petroleum_house_fleet   FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can read travel"   ON petroleum_house_travel  FOR SELECT TO authenticated USING (true);
CREATE POLICY "Auth users can insert fleet"  ON petroleum_house_fleet   FOR INSERT TO authenticated WITH CHECK (true);
CREATE POLICY "Auth users can insert travel" ON petroleum_house_travel  FOR INSERT TO authenticated WITH CHECK (true);

-- Block anonymous access entirely (anon role cannot read or write)
-- (No policies created for anon = blocked by default with RLS enabled)
